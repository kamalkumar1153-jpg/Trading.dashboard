Market may be closed.") return None data = get_market_data() if data: col1, col2 = st.columns(2) with col1: st.subheader("NIFTY 50") st.metric("Price", f"₹ {data['nifty_price']}", f"{data['nifty_change']}%") with col2: st.subheader("SENSEX") st.metric("Price", f"₹ {data['sensex_price']}", f"{data['sensex_change']}%") # RSI Section st.subheader("RSI (14) - NIFTY 50") rsi_color = "🔴 Overbought (Sell Signal)" if data['rsi'] > 70 else \ "🟢 Oversold (Strong Buy Signal)" if data['rsi'] < 30 else \ "⚪ Neutral" st.metric("RSI Value", data['rsi'], rsi_color) # AI Signal st.subheader("🤖 AI SIGNAL") if data['rsi'] > 70: st.error("🔴 **Strong Sell** - Market is Overbought") elif data['rsi'] < 30: st.success("🟢 **Strong Buy** - Market is Oversold") else: st.info("⚪ **Hold** - Neutral Zone, Wait for clearer signal") else: st.warning("Could not fetch live data. Please try again later.") if st.button("🔄 Refresh Data"): st.rerun() st.caption("Data Source: Yahoo Finance | Auto-refresh recommended every 30-60 seconds")
